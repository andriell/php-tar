# php-tarSimple tar writer# Example 1Write tar gz to file    $stringWriter = new FileGzWriter(__DIR__ . '/test.tar.gz');    $tar = new Tar($stringWriter);    $tar->addFile('test1.txt', 'Test String 1');    $tar->addFile('test2.txt', 'Test String 2');    $tar->close();    # Example 2Write tar to file    $stringWriter = new FileWriter(__DIR__ . '/test.tar');    $tar = new Tar($stringWriter);    $tar->addFile('test1.txt', 'Test String 1');    $tar->addFile('test2.txt', 'Test String 2');    $tar->close();# Example 3Write tar gz to string    $stringWriter = new StringGzWriter();    $tar = new Tar($stringWriter);    $tar->addFile('test1.txt', 'Test String 1');    $tar->addFile('test2.txt', 'Test String 2');    $tar->close();    $tarString = $stringWriter->getDate();    # Example 4Write tar to string    $stringWriter = new StringWriter();    $tar = new Tar($stringWriter);    $tar->addFile('test1.txt', 'Test String 1');    $tar->addFile('test2.txt', 'Test String 2');    $tar->close();    $tarString = $stringWriter->getDate();
